# 我的思路
分别写两个函数，用来找符合条件的左边界喝有边界，直接AC了
```java
class Solution {

    public int find_first(int[] nums, int target){

        int left = 0;

        int right = nums.length - 1;

        int mid;

        int ans = -1;

        while(left <= right){

            mid = left + (right - left) / 2;

            if(nums[mid] >= target){

                ans = mid;

                right = mid - 1;

            }else{

                left = mid + 1;

            }

        }

        return ans;

    }

    public int find_last(int[] nums, int target){

        int left = 0;

        int right = nums.length - 1;

        int mid;

        int ans = -1;

        while(left <= right){

            mid = left + (right - left) / 2;

            if(nums[mid] <= target){

                ans = mid;

                left = mid + 1;

            }else{

                right = mid - 1;

            }

        }

        return ans;

    }

    public int[] searchRange(int[] nums, int target) {

        int start = find_first(nums, target);

        if(start == -1 || nums[start] != target){

            return new int[]{-1, -1};

        }

        int end = find_last(nums, target);

        return new int[]{start, end};

    }

}
```
