class Solution {

    public double findMedianSortedArrays(int[] nums1, int[] nums2) {

        // 先用双指针遍历两个数组O(max(num1.length, num2.length))

        List<Integer> list = new ArrayList<>();

        int len1 = nums1.length;

        int len2 = nums2.length;

        int i = 0, j = 0;

        while(i < len1 && j < len2){

            if(nums1[i] < nums2[j]){

                list.add(nums1[i]);

                i ++;

            }else if(nums1[i] == nums2[j]){

                list.add(nums1[i]);

                list.add(nums2[j]);

                i ++;

                j ++;

            }else{

                list.add(nums2[j]);

                j ++;

            }

        }

        while(i < len1){

            list.add(nums1[i]);

            i ++;

        }

        while(j < len2){

            list.add(nums2[j]);

            j ++;

        }

        int total = len1 + len2;

        int left = 0, right = list.size() - 1;

        if(total % 2 == 0){

            int index = total / 2;

            return (double)(list.get(index - 1) + list.get(index)) / 2;

        }else{

            int index = total / 2;

            return (double)(list.get(index));

        }

    }

}
