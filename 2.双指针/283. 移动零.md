# 思路
试了好多次，终于想到了灵魂的一步，`j = i;`降低了时间复杂度，AC这道题目！
```java
class Solution {

    public void moveZeroes(int[] nums) {

        int i = 0, j = 0;

        int len = nums.length;

        while(i < len && j < len){

            // i = 1

            while( i < len && nums[i] != 0){

                i ++;

            }

            // j = 0

            j = i;

            while(j < len && nums[j] == 0){

                j ++;

            }

            if(i <= j  && i < len && j < len){

                int test = nums[i];

                nums[i] = nums[j];

                nums[j] = test;

            }

        }

    }

}
```
